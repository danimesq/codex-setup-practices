ARCH_RAW="$(uname -m)"; case "${ARCH_RAW}" in x86_64|amd64) ARCH="amd64" ;; aarch64|arm64) ARCH="arm64" ;; armv7l|armv6l) ARCH="arm" ;; *) fail "arquitetura nao suportada: ${ARCH_RAW}" ;; esac; get_latest_kubo(){ curl -fsSL https://dist.ipfs.tech/kubo/versions | sed "/^$/d" | grep -E "^v[0-9]+\.[0-9]+\.[0-9]+$" | tail -n 1; }; now="$(date +%s)"; KUBO_VER=""; if [ -f "${CACHE_FILE}" ]; then cache_mtime="$(stat -c %Y "${CACHE_FILE}" 2>/dev/null || stat -f %m "${CACHE_FILE}" 2>/dev/null || echo 0)"; age=$((now-cache_mtime)); if [ "${age}" -lt "${CACHE_TTL}" ]; then KUBO_VER="$(cat "${CACHE_FILE}" || true)"; log "kubo version cache HIT (${KUBO_VER}, age=${age}s)"; fi; fi; if [ -z "${KUBO_VER}" ]; then KUBO_VER="$(get_latest_kubo)"; [ -n "${KUBO_VER}" ] || fail "nao consegui resolver versao do kubo"; printf "%s\n" "${KUBO_VER}" > "${CACHE_FILE}"; log "kubo version cache MISS -> ${KUBO_VER}"; fi; install_kubo(){ if command -v ipfs >/dev/null 2>&1; rm -rf "${WORK}"; mkdir -p "${WORK}"; log "baixando ${URL}"; curl -fL "${URL}" -o "${WORK}/${PKG}" || fail "falha no download do kubo"; tar -xzf "${WORK}/${PKG}" -C "${WORK}" || fail "falha ao extrair kubo"; if [ -x "${WORK}/kubo/install.sh" ]; then log "tentando install.sh oficial"; if (cd "${WORK}/kubo" && bash install.sh); then log "install.sh OK"; else log "install.sh falhou -> fallback binario"; install -m 0755 "${WORK}/kubo/ipfs" "${HOME}/.local/bin/ipfs" || fail "fallback do binario falhou"; fi; else log "install.sh ausente -> fallback binario"; install -m 0755 "${WORK}/kubo/ipfs" "${HOME}/.local/bin/ipfs" || fail "fallback do binario falhou"; fi; command -v ipfs >/dev/null 2>&1 || export PATH="${HOME}/.local/bin:${PATH}"; then log "clonando git-meta em ${TMP}"; rm -rf "${TMP}"; git clone --depth 1 https://github.com/01VCS/git-meta.git "${TMP}" || fail "falha ao clonar git-meta"; else log "reutilizando clone existente em ${TMP}"; command -v git-meta >/dev/null 2>&1 || fail "git-meta nao encontrado no PATH"; }; install_kubo; install_git_meta; log "OK: bootstrap concluido"; log "git-meta: $(command -v git-meta)"; git-meta --help | head -n 2; ls -l .git/hooks/git-meta .git/hooks/pre-commit; ipfs --version'
