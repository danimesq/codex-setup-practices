# 2) Bootstrap do git-meta + kubo (com fallback para evitar quebrar em 403)
bash -c '
set -euo pipefail

REPO="${1:-$(pwd)}"

# Usa diretório atual por padrão para evitar hardcode de caminho que pode não existir em CI/runner.
TMP="/tmp/git-meta"
CACHE_DIR="${HOME}/.cache/kubo"
CACHE_FILE="${CACHE_DIR}/VERSION"
CACHE_TTL="${KUBO_CACHE_TTL_SECONDS:-86400}"
FALLBACK_KUBO_VER="${KUBO_FALLBACK_VERSION:-v0.38.2}"

mkdir -p "${HOME}/.local/bin" "${HOME}/.local/share/git-meta" "${CACHE_DIR}"

log(){ printf "[bootstrap] %s\n" "$*" >&2; }
fail(){ printf "[bootstrap][erro] %s\n" "$*" >&2; exit 1; }

OS="$(uname -s | tr "[:upper:]" "[:lower:]")"
ARCH_RAW="$(uname -m)"
case "${ARCH_RAW}" in
  x86_64|amd64) ARCH="amd64" ;;
  aarch64|arm64) ARCH="arm64" ;;
  armv7l|armv6l) ARCH="arm" ;;
  *) fail "arquitetura nao suportada: ${ARCH_RAW}" ;;
esac

get_latest_kubo(){
  curl -A "codex-bootstrap/1.0" -fsSL https://dist.ipfs.tech/kubo/versions \
    | sed "/^$/d" \
    | grep -E "^v[0-9]+\.[0-9]+\.[0-9]+$" \
    | tail -n 1
}

resolve_kubo_version(){
  if [ -n "${KUBO_VERSION:-}" ]; then
    printf "%s\n" "${KUBO_VERSION}"
    return 0
  fi

  local now cache_mtime age latest
  now="$(date +%s)"

  if [ -f "${CACHE_FILE}" ]; then
    cache_mtime="$(stat -c %Y "${CACHE_FILE}" 2>/dev/null || stat -f %m "${CACHE_FILE}" 2>/dev/null || echo 0)"
    age=$((now-cache_mtime))
    if [ "${age}" -lt "${CACHE_TTL}" ]; then
      latest="$(cat "${CACHE_FILE}" || true)"
      if [ -n "${latest}" ]; then
        log "kubo version cache HIT (${latest}, age=${age}s)"
        printf "%s\n" "${latest}"
        return 0
      fi
    fi
  fi

  latest="$(get_latest_kubo || true)"
  if [ -n "${latest}" ]; then
    printf "%s\n" "${latest}" > "${CACHE_FILE}"
    log "kubo version cache MISS -> ${latest}"
    printf "%s\n" "${latest}"
    return 0
  fi

  log "nao consegui resolver versao mais recente do kubo (ex.: HTTP 403). Usando fallback ${FALLBACK_KUBO_VER}"
  printf "%s\n" "${FALLBACK_KUBO_VER}"
}

KUBO_VER="$(resolve_kubo_version)"

install_kubo(){
  if command -v ipfs >/dev/null 2>&1; then
    log "ipfs ja instalado: $(ipfs --version 2>/dev/null || true)"
    return 0
  fi

  PKG="kubo_${KUBO_VER}_${OS}-${ARCH}.tar.gz"
  URL="https://dist.ipfs.tech/kubo/${KUBO_VER}/${PKG}"
  WORK="/tmp/kubo-install-${KUBO_VER}-${OS}-${ARCH}"

  rm -rf "${WORK}"
  mkdir -p "${WORK}"

  log "baixando ${URL}"
  if ! curl -A "codex-bootstrap/1.0" -fL "${URL}" -o "${WORK}/${PKG}"; then
    log "download do kubo bloqueado/indisponivel; seguindo setup sem ipfs"
    rm -rf "${WORK}"
    return 0
  fi
  if ! tar -xzf "${WORK}/${PKG}" -C "${WORK}"; then
    log "falha ao extrair kubo; seguindo setup sem ipfs"
    rm -rf "${WORK}"
    return 0
  fi

  if [ -x "${WORK}/kubo/install.sh" ]; then
    log "tentando install.sh oficial"
    if (cd "${WORK}/kubo" && bash install.sh); then
      log "install.sh OK"
    else
      log "install.sh falhou -> fallback binario"
      install -m 0755 "${WORK}/kubo/ipfs" "${HOME}/.local/bin/ipfs" || { log "fallback do binario falhou"; rm -rf "${WORK}"; return 0; }
    fi
  else
    log "install.sh ausente -> fallback binario"
    install -m 0755 "${WORK}/kubo/ipfs" "${HOME}/.local/bin/ipfs" || { log "fallback do binario falhou"; rm -rf "${WORK}"; return 0; }
  fi

  command -v ipfs >/dev/null 2>&1 || export PATH="${HOME}/.local/bin:${PATH}"
  if ! command -v ipfs >/dev/null 2>&1; then
    log "ipfs nao encontrado apos tentativa de install; seguindo sem ipfs"
    rm -rf "${WORK}"
    return 0
  fi
  ipfs --version || { log "ipfs instalado, mas comando falhou; seguindo"; rm -rf "${WORK}"; return 0; }
  rm -rf "${WORK}"
}

install_git_meta(){
  if [ ! -d "${TMP}/.git" ]; then
    log "clonando git-meta em ${TMP}"
    rm -rf "${TMP}"
    if ! git clone --depth 1 https://github.com/01VCS/git-meta.git "${TMP}"; then
      log "falha ao clonar git-meta (rede/proxy). Rode novamente em ambiente com acesso ao GitHub"
      return 0
    fi
  else
    log "reutilizando clone existente em ${TMP}"
  fi

  install -m 0755 "${TMP}/git-meta" "${HOME}/.local/bin/git-meta"
  install -m 0755 "${TMP}/init.sh" "${HOME}/.local/share/git-meta/init.sh"
  install -m 0755 "${TMP}/pre-commit" "${HOME}/.local/share/git-meta/pre-commit"
  install -m 0644 "${TMP}/README.md" "${HOME}/.local/share/git-meta/README.md"

  if ! cd "${REPO}"; then
    log "repo nao encontrado: ${REPO}; pulando instalacao dos hooks"
    return 0
  fi
  ln -sf "${HOME}/.local/bin/git-meta" .git/hooks/git-meta
  install -m 0755 "${TMP}/pre-commit" .git/hooks/pre-commit

  if ! command -v git-meta >/dev/null 2>&1; then
    log "git-meta nao encontrado no PATH apos instalacao"
    return 0
  fi
}

install_kubo
install_git_meta

log "OK: bootstrap concluido"
if command -v git-meta >/dev/null 2>&1; then
  log "git-meta: $(command -v git-meta)"
  git-meta --help | head -n 2
  if [ -e .git/hooks/git-meta ] || [ -e .git/hooks/pre-commit ]; then
    ls -l .git/hooks/git-meta .git/hooks/pre-commit 2>/dev/null || true
  fi
else
  log "git-meta indisponivel neste ambiente (provavel bloqueio de rede/proxy)"
fi

if command -v ipfs >/dev/null 2>&1; then
  ipfs --version
else
  log "ipfs indisponivel neste ambiente (setup do git-meta concluido mesmo assim)"
fi
'
